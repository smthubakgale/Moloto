function createModel() 
            {
                let md = tf.sequential();
                const hidden = tf.layers.dense({
                  units: 15,
                  inputShape: [2],
                  activation: 'sigmoid'
                });
              
                const output = tf.layers.dense({
                  units: 9,
                  activation: 'softmax'
                });
                md.add(hidden);
                md.add(output);
              
                const LEARNING_RATE = 0.25;
                const optimizer = tf.train.sgd(LEARNING_RATE);
              
                md.compile({
                  optimizer: optimizer,
                  loss: 'categoricalCrossentropy',
                  metrics: ['accuracy'],
                });
              
                return md;
            }
